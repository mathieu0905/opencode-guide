Config 配置系统

源码位置：packages/opencode/src/config/config.ts

概述

OpenCode 使用四层配置深度合并策略。配置文件名为 opencode.json， 可以放在系统级、用户级、项目级、目录级不同位置。层级越低（越具体），优先级越高。 所有配置都经过 Zod Schema 运行时校验。

Deep Merge 规则


  ═══════ Deep Merge vs Shallow Merge ═══════

  Shallow Merge (浅合并, ❌ 不是 OpenCode 的方式):
    base:    { permission: { bash: "ask", edit: { "*": "allow" } } }
    overlay: { permission: { bash: "allow" } }
    结果:    { permission: { bash: "allow" } }
    → edit 配置丢失了!

  Deep Merge (深合并, ✓ OpenCode 的方式):
    base:    { permission: { bash: "ask", edit: { "*": "allow" } } }
    overlay: { permission: { bash: "allow" } }
    结果:    { permission: { bash: "allow", edit: { "*": "allow" } } }
    → edit 配置保留! bash 被覆盖

  ═══════ 完整示例 ═══════

  用户配置 (~/.config/opencode/opencode.json):
  ┌────────────────────────────────────┐
  │ {                                  │
  │   "provider": {                    │
  │     "anthropic": {                 │
  │       "api_key": "sk-user-key"     │
  │     }                              │
  │   },                               │
  │   "permission": {                  │
  │     "bash": "ask",                 │
  │     "edit": { "*": "allow" }       │
  │   },                               │
  │   "default_model": "anthropic/     │
  │     claude-opus-4"                 │
  │ }                                  │
  └────────────────────────────────────┘
           │ deep merge
           ▼
  项目配置 (.opencode/opencode.json):
  ┌────────────────────────────────────┐
  │ {                                  │
  │   "permission": {                  │
  │     "bash": "allow",              │  ← 覆盖用户配置的 "ask"
  │     "edit": { ".env*": "deny" }    │  ← 追加到 edit 规则
  │   },                               │
  │   "default_model": "anthropic/     │  ← 覆盖用户配置
  │     claude-sonnet-4",              │
  │   "plugin": ["my-plugin@1.0"]      │  ← 项目独有
  │ }                                  │
  └────────────────────────────────────┘
           │
           ▼
  合并结果:
  ┌────────────────────────────────────┐
  │ {                                  │
  │   "provider": {                    │
  │     "anthropic": {                 │  ← 来自用户配置
  │       "api_key": "sk-user-key"     │
  │     }                              │
  │   },                               │
  │   "permission": {                  │
  │     "bash": "allow",              │  ← 项目配置覆盖
  │     "edit": {                      │

  │       ".env*": "deny"             │  ← 来自项目配置 (追加)
  │     }                              │
  │   },                               │
  │   "default_model": "anthropic/     │  ← 项目配置覆盖
  │     claude-sonnet-4",              │
  │   "plugin": ["my-plugin@1.0"]      │  ← 来自项目配置
  │ }                                  │

完整配置 Schema

// opencode.json — 所有可配置项

{
  // ═══════ Provider 配置 ═══════
  "provider": {
    "anthropic": {
      "api_key": "sk-ant-...",     // API Key (也可用环境变量)
      "base_url": "https://...",   // 自定义 API 端点
      "headers": { }               // 自定义请求头
    },
    "openai": { "api_key": "sk-..." },
    // 支持 15+ Provider (见 Provider 章节)
  },

  // ═══════ Model 配置 ═══════
  "model": {
    "claude-opus-4": {              // 覆盖内置模型参数
      "context_window": 200000,
      "max_output_tokens": 16384,
      "cost": { "input": 15.0, "output": 75.0, "cache": 1.5 }
    },
    "my-model": {                   // 自定义模型
      "provider": "openai",
      "id": "my-fine-tuned-model",
      "context_window": 128000
    }
  },

  // ═══════ 默认模型 ═══════
  "default_model": "anthropic/claude-opus-4",

  // ═══════ Agent 配置 ═══════
  "agent": {
    "my-agent": {
      "name": "My Custom Agent",
      "description": "Specialized for specific tasks",
      "mode": "primary",          // "primary" | "subagent" | "all"
      "model": "anthropic/claude-sonnet-4",
      "permission": {
        "edit": { "*.ts": "allow" },
        "bash": "ask"
      }
    }
  },
  "default_agent": "default",

  // ═══════ LSP 配置 ═══════
  "lsp": {
    "typescript": {
      "disabled": false,
      "extensions": [".ts", ".tsx"],
      "command": ["typescript-language-server", "--stdio"],
      "env": { "NODE_ENV": "production" },
      "initialization": { "preferences": { } }
    },
    "python": { "command": ["pyright", "--stdio"] }
  },

  // ═══════ 权限配置 ═══════
  "permission": {
    "*": "allow",
    "bash": "ask",
    "edit": { "*": "allow", ".env*": "deny" },
    "external_directory": "ask"
  },

  // ═══════ 插件 ═══════
  "plugin": [
    "opencode-anthropic-auth@0.0.13",  // NPM 包
    "file:///path/to/local/plugin.js"  // 本地文件
  ],

  // ═══════ 压缩配置 ═══════
  "compaction": {
    "reserved": 20000               // 预留 token 空间 (默认 20K)
  },

  // ═══════ 共享配置 ═══════
  "share": "auto",                  // "auto" | "manual" | "disabled"

  // ═══════ 文件监听 ═══════
  "watcher": {
    "ignore": ["*.log", "tmp/**"]
  },

  // ═══════ 实验性功能 ═══════
  "experimental": {
    "lsp": true,
    "batch": true,
    "filewatcher": true,
    "plan_mode": true
  }
}

配置加载时序


  OpenCode 启动
       │
       ├── 1. 检测运行环境 (OS, 路径)
       │
       ├── 2. 加载配置文件 (按优先级)
       │       ├── /etc/opencode/opencode.json
       │       ├── ~/.config/opencode/opencode.json
       │       ├── {git_worktree}/.opencode/opencode.json
       │       └── {cwd}/.opencode/opencode.json
       │
       ├── 3. Deep merge 所有层
       │
       ├── 4. 环境变量覆盖
       │
       ├── 5. Zod schema 校验
       │       ├── 通过 → 继续
       │       └── 失败 → 报错 + 退出
       │
       ├── 6. Plugin.config hook
       │       └── 插件可以动态修改配置
       │
       └── 7. 最终配置对象 → 全局可用

设计要点

  • 四层合并满足所有场景 — 企业 IT 管控、个人偏好、项目特定、 monorepo 子目录，每一层都有对应的配置位置。
  • Deep Merge 而非覆盖 — 嵌套对象递归合并，不会因为设置一个字段而丢失其他字段。
  • Zod 运行时校验 — 配置加载时通过 Zod schema 验证，无效配置会给出明确错误 （哪个字段、什么问题），而不是运行时莫名其妙地出错。
  • 环境变量最终覆盖 — API Key 等敏感信息不该写在配置文件中（会进 Git）， 环境变量是最安全的方式，且优先级最高。
  • TUI 配置独立 — 终端 UI 配置与核心配置分离， 避免 UI 偏好影响核心功能。
