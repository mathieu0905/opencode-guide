async function truncateOutput(text, options, agent) {
  const lines = text.split("\n")

  // 在限制内 → 直接返回
  if (lines.length <= MAX_LINES && Buffer.byteLength(text) <= MAX_BYTES) {
    return { content: text, truncated: false }
  }

  const preview = []
  let bytes = 0

    const size = Buffer.byteLength(line)
    if (bytes + size > MAX_BYTES || preview.length >= MAX_LINES) break
    preview.push(line)
    bytes += size
  }

  }
