async function loop(sessionID: string) {
  while (true) {
    // ═══════ 第 1 步: 加载消息 ═══════
    // filterCompacted() 只返回最后一次压缩之后的消息
    // 如果消息 10 条中第 5 条有 CompactionPart,
    // 只返回 5-10 (跳过被压缩的历史)
    const messages = await MessageV2.filterCompacted(
      MessageV2.stream(sessionID)
    )

    // ═══════ 第 2 步: 分析当前状态 ═══════
    const lastUser = messages.findLast(m => m.info.role === "user")
    const lastAssistant = messages.findLast(m => m.info.role === "assistant")
    const lastFinished = messages.findLast(m =>
      m.info.role === "assistant" && m.info.finish
    )

    // ═══════ 第 3 步: 退出条件判断 ═══════
    // 如果最新的已完成 AI 消息比最新用户消息更新 → 说明已处理完
    if (lastFinished && lastFinished.info.id < lastUser.info.id) break

    // ═══════ 第 4 步: 处理待执行子任务 ═══════
    // SubtaskPart 表示 Task 工具创建了子 Agent
    // 需要等子 Agent 完成后继续
    const pendingSubtask = findPendingSubtask(messages)
    if (pendingSubtask) {
      await executeSubtask(pendingSubtask)
      continue  // 重新检查消息状态
    }

    // ═══════ 第 5 步: 处理待执行压缩 ═══════
    const pendingCompaction = findPendingCompaction(messages)
    if (pendingCompaction) {
      await SessionCompaction.process({
        messages, sessionID,
        auto: pendingCompaction.auto,
      })
      continue
    }

    // ═══════ 第 6 步: 检查是否需要压缩 ═══════
    // 如果上一轮 token 用量已接近 context window
    if (lastFinished && !lastFinished.summary) {
      if (SessionCompaction.isOverflow({
        tokens: lastFinished.tokens,
        model: currentModel,
      })) {
        await SessionCompaction.create({
          sessionID, auto: true,
          agent: currentAgent.name,
          model: currentModel,
        })
        continue
      }
    }

    // ═══════ 第 7 步: 正常处理 — 核心路径 ═══════
    // 7a. 组装工具 (权限包装 + 插件 Hook)
    const tools = await resolveTools({
      agent, model, session,
      processor, messages,
    })

    // 7b. 创建 AssistantMessage
    const assistant = await MessageV2.create({
      role: "assistant",
      sessionID,
      parentID: lastUser.info.id,
      modelID, providerID, agent: agent.name,
    })

    // 7c. 注入 Reminder (计划模式、构建模式等)
    const systemReminders = await insertReminders(messages)

    // 7d. 构建 System Prompt
    const system = await buildSystemPrompt(agent, model)

    const modelMessages = MessageV2.toModelMessages(messages, model)

    const processor = SessionProcessor.create({

      sessionID, model, abort,
    })

      messages: modelMessages,
      toolChoice: "auto",

        continue
      case "stop":
        return assistant  // 完成
    }
  }

class SessionProcessor {
  private blocked = false           // 权限被拒绝
  private needsCompaction = false   // 需要压缩
  private lastToolParts: ToolPart[] = []  // 用于 Doom Loop 检测

  async process(streamInput): Promise<"compact" | "stop" | "continue"> {

    // 启动 LLM 流
    const stream = LLM.stream(streamInput)

    for await (const event of stream.fullStream) {
      switch (event.type) {

        // ═══════ 文本流处理 ═══════
        case "text-start":
          textPart = await createPart("text", { text: "" })
          break

        case "text-delta":
          textPart.text += event.textDelta
          // ★ Delta 事件: 只发送新增的字符
          Bus.publish("message.part.delta", {
            messageID, partID: textPart.id,
            field: "text",
            delta: event.textDelta  // e.g. "Hello"
          })
          break

        case "text-end":
          await savePart(textPart)  // 持久化最终文本
          break

        // ═══════ 推理过程 (Extended Thinking) ═══════
        case "reasoning-start":
          // 推理有 provider-specific ID, 存在 Map 中
          reasoningMap.set(event.id, await createPart("reasoning"))
          break

        case "reasoning-delta":
          const rPart = reasoningMap.get(event.id)
          rPart.text += event.textDelta
          Bus.publish("message.part.delta", {
            field: "text", delta: event.textDelta
          })
          break

        case "reasoning-end":
          const done = reasoningMap.get(event.id)
          done.text = done.text.trim()  // 清理尾部空白
          reasoningMap.delete(event.id)
          break

        // ═══════ 工具调用状态机 ═══════
        case "tool-input-start":
          // LLM 开始生成工具参数 (流式)
          toolPart = await createToolPart({
            tool: event.toolName,
            state: { status: "pending", input: {} },
          })
          break

        case "tool-input-delta":
          // LLM 逐 token 输出工具参数 (目前 no-op)
          break

        case "tool-call":
          // ★ 参数完成, 开始执行
          toolPart.state = { status: "running", input: event.args }
          Bus.publish("message.part.updated", { part: toolPart })

          // ★★★ Doom Loop 检测 ★★★
          this.lastToolParts.push(toolPart)
          if (this.lastToolParts.length >= 3) {
            const last3 = this.lastToolParts.slice(-3)
            const allSame = last3.every(p =>
              p.tool === last3[0].tool &&
              JSON.stringify(p.state.input) ===
              JSON.stringify(last3[0].state.input)
            )
            if (allSame) {
              // 连续 3 次相同调用！让用户决定
              await permission.ask({ permission: "doom_loop" })
            }
          }
          break

        case "tool-result":
          // 工具执行完成
          toolPart.state = {
            status: "completed",
            input: event.args,
            output: event.result.output,
            title: event.result.title,
            metadata: event.result.metadata,
            attachments: event.result.attachments,
          }
          Bus.publish("message.part.updated", { part: toolPart })
          break

        case "tool-error":
          // 权限拒绝 → 标记 blocked, 稍后终止循环
          if (event.error instanceof PermissionRejected) {
            this.blocked = true
          }
          toolPart.state = { status: "error", error: event.error.message }
          break

        // ═══════ 步骤边界 ═══════
        case "start-step":
          await createPart("step-start")
          // ★ 在每步开始时拍 Git 快照
          snapshot = await Snapshot.track()
          break

        case "finish-step":
          // 记录 token 用量和成本
          await createPart("step-finish", {
            tokens: event.usage,
            cost: calculateCost(event.usage, model),
            finish: event.finishReason,
          })

          // ★ 检查文件变更, 生成 Patch
          const patch = await Snapshot.patch(snapshot)
          if (patch) await createPart("patch", patch)

          // ★★ 溢出检测在 finish-step 之后 ★★
          if (SessionCompaction.isOverflow({
            tokens: event.usage, model,
          })) {
            this.needsCompaction = true
          }
          break

        // ═══════ 错误处理 ═══════
        case "error":
          const parsed = MessageV2.fromError(event.error, { providerID })
          // 分类:
          // - AbortedError → 用户中止
          // - AuthError → API Key 无效
          // - ContextOverflowError → 上下文溢出
          // - APIError(isRetryable) → 可重试
          // - APIError(statusCode=529) → 服务过载
          if (parsed.isRetryable) {
            await createPart("retry")
            return "continue"  // 重试
          }
          break
      }

      // 溢出时提前终止流处理
      if (this.needsCompaction) break
    }

    // ═══════ 返回决策 ═══════
    if (this.needsCompaction) return "compact"
    if (this.blocked || this.error) return "stop"
    if (hasToolCalls) return "continue"  // 有工具调用 → 继续
    return "stop"  // 纯文本 → 结束
  }

function toModelMessages(
  input: WithParts[],
  model: Provider.Model
): ModelMessage[] {

  // ★ 问题 1: 不同 Provider 对工具结果中的媒体支持不同
  const supportsMediaInToolResults = (() => {
    if (model.api.npm === "@ai-sdk/anthropic") return true
    if (model.api.npm === "@ai-sdk/openai") return true
    if (model.api.npm === "@ai-sdk/amazon-bedrock") return true
    if (model.api.npm === "@ai-sdk/google-vertex/anthropic") return true
    // Google Gemini: 只有 3.x 版本支持
    if (model.api.npm === "@ai-sdk/google") {
      return model.api.id.includes("gemini-3")
    }
    return false
  })()

  for (const msg of input) {
    if (msg.info.role === "user") {
      // 用户消息转换
      for (const part of msg.parts) {
        if (part.type === "text" && !part.ignored)
          → { type: "text", text: part.text }

        if (part.type === "file" && part.mime !== "text/plain")
          → { type: "file", url, mediaType, filename }

        // ★ CompactionPart 转换为普通文本
        if (part.type === "compaction")
          → { type: "text", text: "What did we do so far?" }

        // ★ SubtaskPart 转换为文本标记
        if (part.type === "subtask")
          → { type: "text", text: "The following tool was executed by the user" }
      }
    }

    if (msg.info.role === "assistant") {
      // ★ 跳过有错误且没有实质内容的消息
      if (msg.info.error && !hasSubstantiveParts(msg)) continue

      // ★ 检测是否跨模型 (当前模型 vs 生成消息的模型)
      const differentModel = currentModel !== msg.info.modelID

      const media = []  // 待注入的媒体文件

      for (const part of msg.parts) {
        if (part.type === "tool" && part.state.status === "completed") {
          // ★ 已压缩的工具结果 → 替换为占位文本
          const output = part.state.time.compacted
            ? "[Old tool result content cleared]"
            : part.state.output

          // ★ Provider 不支持媒体 → 提取到单独的 user message
          if (!supportsMediaInToolResults) {
            const imageAttachments = part.state.attachments
              .filter(a => a.mime.startsWith("image/"))
            media.push(...imageAttachments)
          }

          → {
            type: "tool-{toolName}",
            state: "output-available",
            toolCallId: part.callID,
            input: part.state.input,
            output,
            // ★ 同模型时传递 providerMetadata (用于缓存)
            ...(differentModel ? {} : { callProviderMetadata: part.metadata })
          }
        }

        // ★ 处理中断的工具调用
        if (part.state.status === "pending" || part.state.status === "running") {
          → {
            type: "tool-{toolName}",
            state: "output-error",
            toolCallId: part.callID,
            errorText: "[Tool execution was interrupted]"
            // ★ Anthropic 要求每个 tool_use 必须有对应的 tool_result
            // 所以中断的调用也要生成 error result
          }
        }
      }

      // ★ 为不支持媒体的 Provider 注入单独的 user message
      if (media.length > 0) {
        result.push({
          role: "user",
          parts: [
            { type: "text", text: "Attached image(s) from tool result:" },
            ...media.map(a => ({ type: "file", url: a.url, mediaType: a.mime }))
          ]
        })
      }
    }
  }

  // 最终转换: UIMessage → AI SDK CoreMessage
  return convertToModelMessages(result, { tools })

// ═══════ 溢出检测算法 ═══════
function isOverflow(tokens, model) {
  // 总 token = input + output + cache.read + cache.write
  const count = tokens.total || sum(tokens)

  // 预留空间 (可配置, 默认 = min(20K, maxOutputTokens))
  const reserved = config.compaction?.reserved
    ?? Math.min(COMPACTION_BUFFER, model.limit.output)

  // 可用空间 = context_window - maxOutputTokens - reserved
  const usable = model.limit.input
    ? model.limit.input - reserved
    : model.limit.context - model.limit.output - reserved

  return count >= usable  // true = 需要压缩
}

async function prune(sessionID) {
  // 从后往前遍历消息
  // ★ 跳过最近 2 轮用户消息 (保持近期上下文新鲜)
  let userTurns = 0
  let accumulated = 0

  for (const msg of messages.reverse()) {
    if (msg.info.role === "user") userTurns++
    if (userTurns <= 2) continue  // 跳过最近 2 轮

    // 如果遇到已有 summary 的 assistant → 已经压缩过, 停止
    if (msg.info.role === "assistant" && msg.info.summary) break

    for (const part of msg.parts) {
      if (part.type !== "tool") continue
      if (part.state.status !== "completed") continue
      if (PRUNE_PROTECTED_TOOLS.includes(part.tool)) continue
      if (part.state.time.compacted) continue  // 已裁剪过

      // 累积要裁剪的 token 量
      accumulated += estimateTokens(part.state.output)

      // 累积到 PRUNE_PROTECT (40K) 才开始标记裁剪
      if (accumulated < PRUNE_PROTECT) continue

      // ★ 标记为已压缩 (不删除数据, 下次 toModelMessages 时替换为占位文本)
      part.state.time.compacted = Date.now()
    }
  }

  // 只有累积量 > PRUNE_MINIMUM (20K) 才真正执行裁剪
  if (accumulated < PRUNE_MINIMUM) return // 不值得裁剪
}
