  return {
    // ═══════ 监听事件 ═══════
    event({ event }) {
      if (event.type === "session.created") {
        console.log("New session:", event.properties.info.id)
      }
      if (event.type === "message.updated") {
        // 可以发送通知、写日志等
      }
    },

    // ═══════ 提供自定义工具 ═══════
    tool: {
      "deploy": {
        description: "Deploy the application",
        parameters: z.object({
          environment: z.enum(["staging", "production"]),
        }),
        execute: async (args, ctx) => {
          // 工具可以请求权限
          await ctx.ask({
            permission: "deploy",
            patterns: [args.environment],
          })

          // 执行部署
          const result = await $`deploy.sh ${args.environment}`.text()

          return {
            title: `Deploy to ${args.environment}`,
            metadata: { env: args.environment },
            output: result,
          }
        },
      },
    },

    // ═══════ 修改 System Prompt ═══════
    async "experimental.chat.system.transform"(input) {
      return {
        system: [
          ...input.system,
          `Project: ${project.name}`,
          "Always respond in Chinese.",
        ]
      }
    },

    // ═══════ 自定义认证 ═══════
    async auth(input) {
      if (input.provider === "my-private-llm") {
        const token = await fetchOAuthToken()
        return { api_key: token }
      }
    },

    // ═══════ 注入环境变量 ═══════
    async "shell.env"({ cwd }) {
      return {
        env: { PROJECT_ROOT: directory }
      }
    },
  }
